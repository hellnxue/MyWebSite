/*导航容器*/
.custom-head-line{
	position:relative;
	width:100%;
	padding:10px 0;
	background-color:#fff;
	
}

/*导航线*/
 .custom-line {
    background-color: #ccc;
    height: 2px;
    left: 10%;
    position: absolute;
    top: 35%;
    width: 80%;
	z-index:1;/*设置好层的顺序*/
} 

/*弹性容器*/
.custom-head-line ul {
	position:relative;
	z-index:10;/*让ul层在导航线的上方*/
    display: flex;/*设置ul为弹性盒子容器*/
    flex-direction: row;/*弹性子元素的排列方式--行排列*/
    justify-content: space-around;/*设置弹性盒子子元素在主轴(横轴)方向上的对齐方式--项目位于各行之前、之间、之后都留有空白的容器内。*/
	align-items: center;/*设置弹性盒子子元素在侧轴（纵轴）方向上的对齐方式--垂直居中*/
	flex-wrap:nowrap;/*设置弹性盒子的子元素超出父元素容器时是否换行--不换行*/
	align-content:center;/*修改 flex-wrap 属性的行为，类似 align-items, 但不是设置子元素对齐，而是设置行对齐*/
	margin:0;
	padding:10px 0;
	 
}

/*弹性子元素*/
.custom-head-line ul li{
	list-style:none;
	flex-grow:1;/*flex-grow用于检索或设置弹性子元素的扩展比率，值为一个数字或initial或inherit*/
	 
  
}

/*可点击区域*/
.custom-head-line ul li a{
	display:block;
	width:70%;
	margin:0 auto;
  	background-color:#fff;
	text-decoration:none;
	text-align:center;
	font-size:20px;
}

.custom-head-line ul li a span{
	display:block;
  	 
}

.custom-head-line ul li a span:last-child{
	font-size:16px;
	margin-top:10px;
  	 
}
